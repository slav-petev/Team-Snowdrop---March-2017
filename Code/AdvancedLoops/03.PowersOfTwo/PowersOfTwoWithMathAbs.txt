using System;

namespace _03.PowersOfTwo
{
    class PowersOfTwoExercise
    {
        static void Main(string[] args)
        {
            var number = int.Parse(
                Console.ReadLine());

            for (var i = 0;
                i <= number;
                i++)
            {
                Console.WriteLine(
					Math.Pow(2, i));
            }
        }
    }
}
